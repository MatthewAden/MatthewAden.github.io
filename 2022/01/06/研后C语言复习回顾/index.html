研后C语言复习回顾

发表于2022-01-06|更新于2022-01-18|C/C++
|字数总计:653|阅读时长:2分钟

关系运算

  • 判断是否相等和不等的优先级比其他关系运算符优先级要低
  • 关系运算符优先级比算术运算符低 比赋值运算符高
  • 小tips： >= 表示的是大于或等于

判断语句

  • if
  • if else
  • 嵌套if语句(可if里嵌套if或if else或if else if else)
  • switch语句 （看作是一种基于计算的跳转，计算控制表达式的值后，程序会跳转到相匹配的case,如果执行完分支后，没有遇到break会继续执行下一个分支，这一点尤为要注意）
  • switch的控制表达式必须要是整型
  • case(常量) 常量也可以是常量表达式 例如1+1

简单代码练习

  • 小启示 尽量写单一出口的代码
1.
// 英尺换算
// 思路：
#include <stdio.h>
int main()
{
    int cm, foot, inch;
    scanf("%d", &cm);
    foot = cm / 30.48; //强制类型转换,会将结果由浮点数转换整数
    inch = (cm / 30.48 - foot) * 12;
    printf("%d %d", foot, inch);
    return 0;
}

//然后是几点
//思路：输入一个时间 首先利用除法和取余得到时和分钟 其次加上流逝的分钟数 最后再用取余和除法转换成终止时间
include <stdio.h>
int main()
{
    int begin_time, end_time, time;
    scanf("%d %d", &begin_time, &time);
    end_time = begin_time / 100 * 60 + begin_time % 100 + time;
    printf("%d%d", end_time / 60, end_time % 60);
    return 0;
}


3.
// 逆序的三位数且输入数字含有结尾0时输出不应带有前导的0，比如输入700，输出应为7
// 思路：利用取余、除法运算分离百十个位，同时利用判断语句实现除0要求
#include <stdio.h>
int main()
{
    int number, i, j, k;
    scanf("%d", &number);
    i = number / 100;      //百位
    j = number % 100 / 10; //十位
    k = number % 100 % 10; //个位
    if (k == 0 & j == 0)
    {
        printf("%d", i);
    }
    else if (k == 0 & j != 0)
    {
        printf("%d%d", j, i);
    }
    else
    {
        printf("%d%d%d", k, j, i);
    }
    return 0;
}

// 成绩转换
#include <stdio.h>
int main()
{
    int score;
    scanf("%d", &score);
    score /= 10;
    switch (score)
    {
    case 10:      // switch执行完一条分支的语句后如不遇到break会继续顺序执行下一个分支
    case 9:
        printf("A\n");
        break;
    case 8:
        printf("B\n");
        break;
    case 7:
        printf("C\n");
        break;
    case 6:
        printf("D\n");
        break;
    default:
        printf("E\n");
        break;
    }
    return 0;
}
